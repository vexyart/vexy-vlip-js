import { VexyVlip } from "vexy-vlip-js";
// (in this demo: from the built bundle "./vexy-vlip.element.js")

const vlip = new VexyVlip(document.getElementById("host"), {
  src: "media/sample.mp4",
  track: "media/sample.vtt",
  mode: "stepped",        // or "continuous"
  easing: "ease-in-out",  // optional eased travel between cards
  muted: true,
});

vlip.next();  vlip.prev();  vlip.setMode("continuous");
host.addEventListener("vexyvlip:stop", (e) => console.log(e.detail.index));
